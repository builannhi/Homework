Viết hàm USD2VND(n), hàm nhận một tham số (input) n là số tiền USD, tính và in ra số tiền tương ứng theo mệnh giá Việt Nam theo công thức:

VND = USD * 23.500

 
let USD2VND = function (n) {
  let m = n * 23500;
  console.log(`${n} USD = ${m} VND`);
};

USD2VND(34);

Tính lãi suất kép ngân hàng

Viết hàm compoundInterest(P, r, e, t), hàm nhận 4 tham số (input):

P: số tiền gốc ban đầu

r: lãi suất hàng năm

e: số lần ghép lãi 1 năm (*)

t: số năm gửi ngân hàng

Công thức: A = P * (1 + r / n)**(n * t)

 
let compoundInterest = function (P, r, t) {
  const e = 12;
  let A = P * (1 + r / e) ** (e * t);
  console.log(
    `If you deposit VND${P}  in a bank with an interest rate of ${r} %/year, after ${t} years you will receive VND${A} `
  );
};

compoundInterest(1000000, 0.6, 0.5);
